CREATE TABLE Customer (
    cxID INT AUTO_INCREMENT PRIMARY KEY,
    cx_Name VARCHAR(255) NOT NULL,
    Email VARCHAR(255) NOT NULL,
    password VARCHAR(255) NOT NULL
);

CREATE TABLE Menu (
    item_ID INT AUTO_INCREMENT PRIMARY KEY,
    item_Name VARCHAR(255) NOT NULL,
    Description TEXT,
    Price DECIMAL(10,2) NOT NULL,
    Category VARCHAR(255) NOT NULL,
    IMG_path VARCHAR(500)
);

CREATE TABLE `Table` (
    TableID INT AUTO_INCREMENT PRIMARY KEY,
    isReserved BOOLEAN NOT NULL,
    chairs_count INT NOT NULL
);

CREATE TABLE Order (
    order_ID INT AUTO_INCREMENT PRIMARY KEY,
    cx_ID INT,
    item_ID INT,
    FOREIGN KEY (cx_ID) REFERENCES Customer(cxID),
    FOREIGN KEY (item_ID) REFERENCES Menu(item_ID)
);

CREATE TABLE Reservation (
    Cx_ID INT,
    Table_ID INT,
    Reservation_Date DATETIME NOT NULL,
    PRIMARY KEY (Cx_ID, Table_ID, Reservation_Date),
    FOREIGN KEY (Cx_ID) REFERENCES Customer(cxID),
    FOREIGN KEY (Table_ID) REFERENCES Table_(Table_ID)
);
